\newcommand\newmathfamily[3]{%
  \def\umathfontname{#1}
  \edef\umathfontdecl{[\umathfontname]:script=math;}
  \DeclareFontFamily{\EUENC}{#2}{}%
  \DeclareFontShape{\EUENC}{#2}{m}{n}
    {<-6>"\umathfontdecl;color=AA0000;+ssty=1;#3"%
     <6-9>"\umathfontdecl;color=00AA00;+ssty=0;#3"%
     <9->"\umathfontdecl;#3"}{}%
}
